    l = []

    l = [0]
else:

    b = l[1]

data = [{'name': 'Alice', 'age': 25}, {'name': 'Bob', 'age': 22}]

sorted_data = sorted(data, key=lambda x: x['age'])

# How The above code of line will work as.
#     1. The sorted(iterable, key=none, value=none) function returns a new sorted list from the elements of any iterable.
#          The key takes a function which is applies to every iterable and the value returned by that function sorting will be done on the basis of that data.
#     2. The lambda function is a small anonymous function. which take x as an argument and return x['age'] as a value.
#          Here, x will be the dict of the data, and x['age'] will return the age of the person.
#     3. So the sorting will be done on the basis of the age of the person.

print(sorted_data)

[{'name': 'Bob', 'age': 22}, {'name': 'Alice', 'age': 25}]

from matplotlib_venn import venn2, venn3
import matplotlib.pyplot as plt

l = set([i for i in range(30) if i % 2 == 0])
l1 = set([i for i in range(20)])
# I want to find union of l and l1
union = l |  l1 # | -> this is used for union
intersection = l & l1 # & -> this is used for intersection
sym_difference = l ^ l1 # - -> this is used for symmetric difference.

print("Union: ", len(union))
print("Intersection: ", len(intersection))
print("Symmetric Difference: ", len(sym_difference))

plt.figure(figsize=(8, 6)) # This is used to determine the sixe of the figure.
venn = venn2([set(l), set(l1)], ('Set 1', 'Set 2'))  # Labels for sets
plt.title("Venn Diagram for 2 Sets")

# Annotate the results in the diagram
venn.get_label_by_id('10').set_text('\n'.join(map(str, l - l1)))  # Elements in section 1
# Here 10 refers to the elements which are only in Set 1 (set ma ha or set 2 ma nhi)
venn.get_label_by_id('01').set_text('\n'.join(map(str, l1 - l)))  # Elements only in Set 2
# Here 01 refers to the elements which are only in Set 2 (set2 ma ha or set1 ma nhi)
venn.get_label_by_id('11').set_text('\n'.join(map(str, intersection)))  # Intersection
# Here 11 refers to the elements which are in both the sets (set1 ma ha or set2 ma ha)

# Add title
plt.title("Venn Diagram of Set Operations")

plt.show()
